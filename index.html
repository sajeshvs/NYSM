        with tf.GradientTape() as tape:
            outputs = model(inputs)
            loss = outputs.loss
        gradients = tape.gradient(loss, model.trainable_variables)
        optimizer.apply_gradients(zip(gradients, model.trainable_variables))

// Process video using FFmpeg
func processVideo(inputPath string, outputPath string) error {
    cmd := exec.Command("ffmpeg",
        "-i", inputPath,
        "-c:v", "libx264",
        "-crf", "23",
        "-preset", "medium",
        "-c:a", "aac",
        "-b:a", "128k",
        "-movflags", "+faststart",
        outputPath,
    )
    
    var stderr bytes.Buffer
    cmd.Stderr = &stderr
    
    if err := cmd.Run(); err != nil {
        return fmt.Errorf("ffmpeg error: %v, %s", err, stderr.String())
    }
    return nil
}

@PostMapping("/jobs")
public ResponseEntity createJob(@RequestBody JobRequest jobRequest) {
    // Validate input
    if (!jobValidator.isValid(jobRequest)) {
        throw new InvalidJobException("Invalid job data");
    }
    
    // Map to entity
    Job job = jobMapper.toEntity(jobRequest);
    
    // Save to database
    Job savedJob = jobRepository.save(job);
    
    // Publish job created event
    eventPublisher.publishJobCreatedEvent(savedJob);

            .body(savedJob);
}

// Send notification to user
async function sendNotification(userId, message, type) {
    // Get user preferences
    const preferences = await getUserPreferences(userId);
    
    // Check if user wants this type of notification
    if (!preferences[type]) return;
    
    // Prepare notification channels
    const channels = [];
    if (preferences.inApp) channels.push('inApp');
    if (preferences.email) channels.push('email');
    if (preferences.sms) channels.push('sms');
    if (preferences.push) channels.push('push');
    
    // Send through selected channels
    for (const channel of channels) {
        switch(channel) {
            case 'inApp':
                sendInAppNotification(userId, message);
                break;
            case 'email':
                sendEmailNotification(userId, message);
                break;
            case 'sms':
                sendSmsNotification(userId, message);
                break;
            case 'push':
                sendPushNotification(userId, message);
                break;
        }
    }
    
    // Log notification
    logNotification(userId, message, type, channels);
}

Blockchain Verification:
// Store verification record on blockchain
public async Task StoreVerificationRecord(VerificationRecord record)
{
    // Create transaction
    var transaction = new Transaction
    {
        FromAddress = _config.WalletAddress,
        ToAddress = record.UserAddress,
        Value = 0,
        Data = JsonConvert.SerializeObject(record)
    };
    
    // Sign transaction
    transaction.Sign(_config.PrivateKey);
    
    // Submit to blockchain
    var client = new BlockchainClient(_config.NodeUrl);
    var response = await client.SubmitTransaction(transaction);
    
    if (!response.Success)
        throw new VerificationException("Failed to store verification record");
    
    return response.TransactionHash;
}

// JWT token generation
const generateToken = (user) => {
    return jwt.sign(
        {
            userId: user._id,
            role: user.role,
            email: user.email
        },
        process.env.JWT_SECRET,
        {
            expiresIn: '1h',
            issuer: 'nysm-auth-service'
        }
    );
};

// Token verification middleware
const verifyToken = (req, res, next) => {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: 'Access denied' });
    
    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.user = decoded;
        next();
    } catch (err) {
        res.status(400).json({ error: 'Invalid token' });
    }
};
